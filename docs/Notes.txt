<div class="gallery" id="orgGallery"></div>

<script>
fetch('fetch_organizations.php')
  .then(res => res.json())
  .then(data => {
    const gallery = document.getElementById('orgGallery');
    data.forEach(org => {
      const members = org.members.map(m => `<li>${m}</li>`).join('');
      const card = `
      <label class="card">
        <input type="checkbox" />
        <article>
          <div class="front"><span><img src="uploads/orgs/${org.logo}" alt="${org.name}" /></span></div>
          <div class="back">
            <h4>${org.name}</h4>
            <p>${org.description}</p>
            <ul>${members}</ul>
          </div>
        </article>
      </label>`;
      gallery.innerHTML += card;
    });
  });
</script>
